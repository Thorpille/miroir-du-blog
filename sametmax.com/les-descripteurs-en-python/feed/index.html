class classproperty(object):
    def __init__(self, getter):
        self.getter = getter
 
    def __get__(self, instance, owner):
        return self.getter(owner)

Ensuite, par exemple :

class MyClass(object):
    @classproperty
    def tagname(cls):
        return "data-%s" % cls.__name__
